    protected void initFormLetterReport()
    {
        PurchLine purchLine;
        VendPurchOrderJour orderJour;
        PurchTable purchTable;
        str orderstr = vendPurchOrderJour.PurchId;
        RecordSortedList            recordListLocal;
        PurchPurchaseOrderContract  purchPurchaseOrderContract = contract as PurchPurchaseOrderContract;
        this.parmShowDialog(true);
        if (args.record().TableId == tableNum(VendPurchOrderJour))
        {
            orderJour = args.record();
            purchTable =  orderJour.purchTable();
            select firstonly purchLine
                where purchLine.PurchId == purchTable.PurchId
                && (purchLine.AssetId || purchLine.AssetGroup);
        }
        else
        {
            recordListLocal = this.splitToRecordList();
            recordListLocal.first(orderJour);
            purchTable =  orderJour.purchTable();
            select firstonly purchLine
                    where purchLine.PurchId == purchTable.PurchId
                    && (purchLine.AssetId || purchLine.AssetGroup);
        }
        if (!purchLine)
        {
            this.parmShowDialog(false);
        }

        if (classIdGet(this.parmArgs().caller()) == classNum(PurchPurchOrderJournalPrint))
        {
            purchPurchOrderJournalPrint = this.parmArgs().caller();
        }
        super();
    }
